#put the count into a dict

from collections import defaultdict


clust=defaultdict(dict)

for pair in zip(clusters,classes):
    key_clust=pair[0]
    key_class=pair[1]
    if clust[key_clust]:
        if not key_class in clust[key_clust]:
            clust[key_clust][key_class]=1
        else:
            clust[key_clust][key_class]+=1
    else:
        clust[key_clust]={key_class:1}
